# Root is the country creating the CB.
# scope:target is the repeated Sea Peoples raider.

cb_sea_peoples_punitive_raid = {
	trade = yes
	no_spy_network_cost = yes

	create_visible = {
		scope:target = {
			sea_peoples_raider_country_identity = yes
			has_variable = sea_peoples_raid_infamy
		}
	}

	create_enabled = {
		NOT = { has_truce_with = scope:target }
		scope:target = {
			sea_peoples_raider_country_identity = yes
			var:sea_peoples_raid_infamy >= 3
		}
		OR = {
			num_of_ports > 0
			sea_peoples_crisis_maritime_target = yes
		}
	}

	speed = 12
	war_goal_type = naval
}
