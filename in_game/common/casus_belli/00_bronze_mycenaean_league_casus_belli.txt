# Internal Mycenaean League wars. These mirror the vanilla Japanese clan CBs
# but use Bronze Age palace authority instead of shogunate terminology.

cb_mycenaean_palace_rebel_against_high_wanax = {
	create_visible = {
		is_member_of_international_organization = international_organization:mycenaean_league
		NOT = { is_leader_of_international_organization = international_organization:mycenaean_league }
	}

	declare_enabled = {
		is_member_of_international_organization = international_organization:mycenaean_league
		NOT = { is_leader_of_international_organization = international_organization:mycenaean_league }
	}

	create_enabled = {
		is_member_of_international_organization = international_organization:mycenaean_league
		scope:target = {
			is_leader_of_international_organization = international_organization:mycenaean_league
		}
	}

	speed = 10
	max_warscore_from_battles = 100
	war_goal_type = clan_expansion_wargoal
}
