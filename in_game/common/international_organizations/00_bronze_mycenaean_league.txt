mycenaean_league = {
	has_target = no
	unique = yes
	show_on_diplomatic_map = yes
	expel_members_who_are_targets_of_other_members = no
	show_leave_message = yes
	use_laws_as_join_reason = yes
	has_dynastic_power = yes

	map_color = hsv360 { 36 72 70 }
	leader_color = define:NMapColors|INTERNATIONAL_ORGANIZATION_LEADER_COLOR

	show_as_overlord_on_map_trigger = {
		international_organization_has_leader = yes
	}

	has_leader_country = yes
	leader_title_key = "MYCENAEAN_LEAGUE_LEADER"
	title_is_suffix = no
	override_ruler_title = yes
	leader_type = character
	leader = {
		leader_country ?= {
			if = {
				limit = {
					has_regent = yes
					has_heir = yes
				}
				heir ?= { add_to_list = leaders }
			}
			else_if = {
				limit = {
					has_ruler = yes
					ruler ?= { is_alive = yes }
				}
				ruler ?= { add_to_list = leaders }
			}
		}
	}

	leader_change_trigger_type = timed
	leader_change_method = score
	months_between_leader_changes = 24
	disband_if_no_leader = no

	leader_score = {
		value = great_power_score
		add = {
			desc = MYCENAEAN_LEAGUE_PALACE_WEALTH
			value = monthly_income_trade_and_tax
			multiply = 8
		}
		if = {
			limit = {
				has_special_status_in_international_organization = {
					type = special_status:great_palace_candidate
					international_organization = scope:recipient
				}
			}
			add = {
				desc = MYCENAEAN_LEAGUE_GREAT_PALACE
				value = 350
			}
		}
		if = {
			limit = {
				has_special_status_in_international_organization = {
					type = special_status:palace_candidate
					international_organization = scope:recipient
				}
			}
			add = {
				desc = MYCENAEAN_LEAGUE_SECONDARY_PALACE
				value = 150
			}
		}
		if = {
			limit = {
				has_special_status_in_international_organization = {
					type = special_status:sanctuary_patron
					international_organization = scope:recipient
				}
			}
			add = {
				desc = MYCENAEAN_LEAGUE_SANCTUARY_INFLUENCE
				value = 75
			}
		}
		if = {
			limit = {
				has_special_status_in_international_organization = {
					type = special_status:island_wanaxate
					international_organization = scope:recipient
				}
			}
			subtract = {
				desc = MYCENAEAN_LEAGUE_ISLAND_DISTANCE
				value = 250
			}
			add = {
				desc = MYCENAEAN_LEAGUE_MARITIME_WEALTH
				value = monthly_income_trade_and_tax
				multiply = 20
			}
		}
		if = {
			limit = { has_variable = mycenaean_megaron_council }
			add = {
				desc = MYCENAEAN_LEAGUE_MEGARON_COUNCIL
				value = 85
			}
		}
		if = {
			limit = { has_variable = mycenaean_shared_sacrifice }
			add = {
				desc = MYCENAEAN_LEAGUE_SHARED_SACRIFICE
				value = 65
			}
		}
		if = {
			limit = { has_variable = mycenaean_linear_b_administration }
			add = {
				desc = MYCENAEAN_LEAGUE_LINEAR_B
				value = 55
			}
		}
		if = {
			limit = { has_variable = mycenaean_pan_achaean_expedition }
			add = {
				desc = MYCENAEAN_LEAGUE_PAN_ACHAEAN_EXPEDITION
				value = 120
			}
		}
		if = {
			limit = { has_variable = mycenaean_rival_claimant }
			add = {
				desc = MYCENAEAN_LEAGUE_RIVAL_CLAIMANT
				value = 110
			}
		}
		if = {
			limit = { has_variable = mycenaean_bronze_tribute_demand }
			subtract = {
				desc = MYCENAEAN_LEAGUE_FORCED_TRIBUTE
				value = 70
			}
		}
	}

	land_ownership_rule = mycenaean_league_land_ownership
	antagonism_modifier_for_taking_land_from_fellow_member = 0.75
	antagonism_modifier_for_taking_land_from_member_as_outsider = 1.5
	no_cb_price_modifier_for_fellow_member = 0.75

	# The High Wanax can draw the league into defensive wars when authority is healthy.
	only_leader_country_joins_defensive_wars = no
	join_defensive_wars_auto_call = {
		exists = scope:target
		scope:target ?= {
			NOT = { is_member_of_international_organization = root }
		}
		var:mycenaean_league_authority >= 45
	}
	join_defensive_wars_can_call = {
		exists = scope:target
		scope:target ?= {
			NOT = { is_member_of_international_organization = root }
		}
		var:mycenaean_league_authority >= 20
	}
	join_offensive_wars_can_call = {
		exists = scope:actor
		scope:actor ?= {
			is_leader_of_international_organization = root
		}
		var:mycenaean_league_authority >= 65
	}

	can_declare_war = {
		always = yes
	}

	gives_military_access_to_all_when_at_war = yes
	can_initiate_policy_votes = {
		is_leader_of_international_organization = scope:recipient
	}

	create_visible_trigger = { always = no }
	invite_visible_trigger = { always = yes }

	can_lead_trigger = {
		is_subject = no
		is_rebel_country = no
		is_member_of_international_organization = scope:recipient
	}

	can_join_trigger = {
		is_rebel_country = no
		OR = {
			culture = culture:mycenaean
			any_owned_location = {
				is_owned_by_international_organization = scope:recipient
			}
		}
	}

	can_leave_trigger = {
		scope:recipient = {
			var:mycenaean_league_authority < 25
		}
	}

	auto_leave_trigger = {
		OR = {
			in_civil_war = yes
			scope:recipient = {
				var:mycenaean_league_authority <= 5
			}
		}
	}

	auto_disband_trigger = {
		OR = {
			total_members < 2
			var:mycenaean_league_authority <= 0
		}
	}
	disband_minimum_member_count = 1

	modifier = {
		reject_subjugation_reasons = 10
	}

	leader_modifier = {
		monthly_prestige = 0.1
		diplomatic_reputation = diplomatic_reputation_weak_bonus
		great_power_score_exempt_from_forfeit = 25
		government_size = 1
	}

	non_leader_modifier = {
		monthly_prestige = 0.02
		reject_subjugation_reasons = 10
	}

	monthly_effect = {
		if = {
			limit = { var:mycenaean_league_authority > 100 }
			set_variable = { name = mycenaean_league_authority value = 100 }
		}
		if = {
			limit = { var:mycenaean_league_authority < 0 }
			set_variable = { name = mycenaean_league_authority value = 0 }
		}
	}

	variables = {
		mycenaean_league_authority = {
			format = "MYCENAEAN_LEAGUE_AUTHORITY_DISPLAY"
			change_format = "VARIABLE_CHANGE_FORMAT"
			min = 0
			max = 100
			monthly_change = {
				add = {
					add = {
						desc = MYCENAEAN_LEAGUE_INTERNAL_PEACE
						if = {
							limit = { international_organization_has_internal_peace = yes }
							value = 0.05
						}
						else = {
							value = -0.2
						}
					}
					add = {
						desc = MYCENAEAN_LEAGUE_MEMBER_NETWORK
						value = total_members
						multiply = 0.01
						subtract = 0.06
					}
					if = {
						limit = { international_organization_has_leader = no }
						subtract = {
							desc = MYCENAEAN_LEAGUE_NO_HIGH_WANAX
							value = 0.5
						}
					}
					if = {
						limit = {
							international_organization_has_leader = yes
							leader_country ?= { at_war = yes }
						}
						add = {
							desc = MYCENAEAN_LEAGUE_HIGH_WANAX_AT_WAR
							value = 0.02
						}
					}
					if = {
						limit = { total_members < 5 }
						subtract = {
							desc = MYCENAEAN_LEAGUE_FRAGMENTED_PALACES
							value = 0.25
						}
					}
					if = {
						limit = {
							international_organization_has_leader = yes
							leader_country ?= { has_variable = mycenaean_shared_sacrifice }
						}
						add = {
							desc = MYCENAEAN_LEAGUE_SHARED_SACRIFICE
							value = 0.06
						}
					}
					if = {
						limit = {
							international_organization_has_leader = yes
							leader_country ?= { has_variable = mycenaean_palace_mediation }
						}
						add = {
							desc = MYCENAEAN_LEAGUE_MEDIATED_RIVALRIES
							value = 0.05
						}
					}
					if = {
						limit = {
							international_organization_has_leader = yes
							leader_country ?= { has_variable = mycenaean_bronze_tribute_demand }
						}
						subtract = {
							desc = MYCENAEAN_LEAGUE_FORCED_TRIBUTE
							value = 0.08
						}
					}
					if = {
						limit = {
							international_organization_has_leader = yes
							leader_country ?= { has_variable = mycenaean_palaces_burning }
						}
						subtract = {
							desc = MYCENAEAN_LEAGUE_PALACES_BURNING
							value = 0.35
						}
					}
					if = {
						limit = {
							international_organization_has_leader = yes
							leader_country ?= { has_variable = sea_peoples_internal_breakdown }
						}
						subtract = {
							desc = MYCENAEAN_LEAGUE_COLLAPSE_PRESSURE
							value = 0.18
						}
					}
				}
			}
		}
	}

	ai_desire_to_join = {
		add = {
			desc = MYCENAEAN_LEAGUE_SHARED_CULTURE
			value = 50
			if = {
				limit = { NOT = { culture = culture:mycenaean } }
				multiply = 0.25
			}
		}
	}

	ai_desire_to_allow_new_member = {
		add = {
			desc = MYCENAEAN_LEAGUE_WANTS_PALACE_NETWORK
			value = 25
		}
	}

	special_statuses_implemented = {
		high_wanax
		great_palace_candidate
		palace_candidate
		palace_kingdom
		sanctuary_patron
		island_wanaxate
	}
}
