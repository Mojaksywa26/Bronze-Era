wanaxate_hegemony = {
	type = international_organization
	potential = { international_organization_type = international_organization_type:mycenaean_league }
	allow = { always = yes }
	requires_vote = yes

	equal_palaces_policy = {
		country_modifier = {
			potential_trigger = {
				NOT = { is_leader_of_international_organization = scope:recipient }
			}
			reject_subjugation_reasons = 20
		}
		wants_this_policy_bias = {
			add = { desc = MYCENAEAN_LEAGUE_MEMBER_AUTONOMY value = 25 }
		}
	}

	recognized_high_wanax_policy = {
		country_modifier = {
			potential_trigger = {
				is_leader_of_international_organization = scope:recipient
			}
			monthly_prestige = 0.05
			diplomatic_reputation = diplomatic_reputation_weak_bonus
		}
		country_modifier = {
			potential_trigger = {
				NOT = { is_leader_of_international_organization = scope:recipient }
			}
			reject_subjugation_reasons = 10
		}
		wants_this_policy_bias = {
			add = { desc = MYCENAEAN_LEAGUE_BALANCED_HEGEMONY value = 10 }
		}
	}

	great_wanax_policy = {
		country_modifier = {
			potential_trigger = {
				is_leader_of_international_organization = scope:recipient
			}
			monthly_prestige = 0.1
			diplomatic_reputation = diplomatic_reputation_mild_bonus
			government_size = 1
		}
		join_offensive_wars_can_call = {
			exists = scope:actor
			scope:actor ?= {
				is_leader_of_international_organization = root
			}
			var:mycenaean_league_authority >= 55
		}
		wants_this_policy_bias = {
			subtract = { desc = MYCENAEAN_LEAGUE_MEMBER_AUTONOMY value = 20 }
		}
	}
}
