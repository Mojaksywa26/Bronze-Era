			country_type = building
			NOT = { is_leader_of_international_organization = international_organization:mycenaean_league }
		}
	}

	allow = {
		scope:actor ?= {
			at_war = no
			is_subject = no
			mycenaean_palace_has_valid_landable_location = yes
			OR = {
				situation:mycenaean_fragmentation ?= {
					has_variable = mycenaean_fragmentation_crisis_active
				}
				international_organization:mycenaean_league = {
					var:mycenaean_league_authority <= 25
				}
				AND = {
					has_variable = mycenaean_palace_influence
					var:mycenaean_palace_influence >= 45
				}
			}
		}
		trigger_if = {
			limit = {
				exists = international_organization:mycenaean_league.leader_country
			}
			international_organization:mycenaean_league.leader_country = {
				scope:actor = { NOT = { has_truce_with = prev } }
			}
		}
	}

	effect = {
		scope:actor ?= {
			turn_mycenaean_palace_into_normal_tag = yes
			subtract_mycenaean_league_authority = { value = 10 }
			custom_tooltip = mycenaean_proclaim_palace_autonomy_tt
		}
		international_organization:mycenaean_league.leader_country ?= {
			add_casus_belli = { type = casus_belli:cb_mycenaean_high_wanax_reassert_authority target = scope:actor }
			trigger_event_silently = mycenaean_league.225
		}
	}

	ai_will_do = {
		add = -6
		international_organization:mycenaean_league.leader_country ?= {
			scope:actor = {
				add = {
					value = this_opinion_of_prev
					multiply = -0.1
				}
			}
		}
	}
