a_footmen = {
	default = yes
	copy_from = a_age_1_traditions_heavy_infantry
	strength_damage_taken = -0.05
	
	
	country_potential = { 
		government_type != government_type:steppe_horde
	}

	#root = location. Checks location_potential first so don't need that scripting in
	mercenaries_per_location = {
		pop_type = laborers
		multiply = 0.1
	}
	
	upgrades_to = a_men_at_arms

	
	maintenance_demand =  early_heavy_infantry_maintenance
	construction_demand = early_heavy_infantry_construction
	
	gfx_tags = { medium_tag }
	

}

a_archers = {
	default = yes
	copy_from = a_age_1_traditions_light_infantry
	

	upgrades_to = a_crossbowmen

	
	country_potential = { 
		government_type != government_type:steppe_horde
	}

	#root = location. Checks location_potential first so don't need that scripting in
	mercenaries_per_location = {
		pop_type = peasants
		multiply = 0.05
	}
	
	maintenance_demand = archer_infantry_maintenance
	construction_demand = archer_infantry_construction
	gfx_tags = { archer_tag medium_tag }
}

a_footmen_levy = {
	copy_from = a_footmen
	
	buildable = no
	levy = yes
	gfx_tags = { medium_tag }
}

a_horsemen = {
	copy_from = a_age_1_traditions_light_cavalry
	
	
	upgrades_to = a_cavalrymen

	#root = location. Checks location_potential first so don't need that scripting in
	mercenaries_per_location = {
		pop_type = nobles
		multiply = 0.3
	}
	
	mercenaries_per_location = {
		pop_type = tribesmen
		multiply = 0.01
	}
	gfx_tags = { medium_tag }
}

a_armored_horsemen = {
	copy_from = a_age_1_traditions_heavy_cavalry

	#root = location. Checks location_potential first so don't need that scripting in
	mercenaries_per_location = {
		pop_type = nobles
		multiply = 0.1
	}
	mercenaries_per_location = {
		pop_type = burghers
		multiply = 0.05
	}
	
	maintenance_demand =  heavy_cavalry_maintenance
	construction_demand = heavy_cavalry_construction

	upgrades_to = a_heavy_cavalrymen

	gfx_tags = { heavy_tag }
}

a_camp_followers = {
	copy_from = a_age_1_traditions_auxiliary
	gfx_tags = { medium_tag }
	
	country_potential = { 
		and = {
			exists = government_type
			government_type != government_type:steppe_horde
		}
	}

	#root = location. Checks location_potential first so don't need that scripting in
	mercenaries_per_location = {
		pop_type = clergy
		multiply = 0.01
	}
	
	upgrades_to = a_supply_carts

}
