mycenaean_palace_tribute = {
	get_payer_list = {
		every_international_organization_member = {
			limit = {
				NOT = { is_leader_of_international_organization = root }
			}
			add_to_local_variable_list = {
				name = payers
				target = this
			}
		}
	}

	get_payee_list = {
		leader_country ?= {
			add_to_local_variable_list = {
				name = payees
				target = this
			}
		}
	}

	price = mycenaean_palace_tribute_price
	uses_maintenance = yes
	min_slider_value = 0.25

	maintenance_modifier = {
		potential_trigger = {
			"payment_maintenance(scope:recipient|mycenaean_palace_tribute)" < 0.5
		}
		scale = {
			value = 1
			subtract = {
				value = "payment_maintenance(scope:recipient|mycenaean_palace_tribute)"
				multiply = 2
			}
		}
		monthly_prestige = -0.05
	}

	price_multiplier = {
		if = {
			limit = {
				international_organization_has_leader = no
			}
			value = 0
		}
		else = {
			every_international_organization_member = {
				limit = {
					NOT = { is_leader_of_international_organization = root }
				}
				add = {
					desc = MYCENAEAN_LEAGUE_TRIBUTE_BASE
					value = monthly_income_trade_and_tax
					multiply = 0.06
				}
				add = {
					desc = MYCENAEAN_LEAGUE_TRIBUTE_STORED_WEALTH
					value = gold
					multiply = 0.0005
				}
				if = {
					limit = { mycenaean_league_has_palace_house = yes }
					add = {
						desc = MYCENAEAN_LEAGUE_PALACE_HOUSE_TRIBUTE
						value = 0.25
					}
				}
			}
		}
	}

	proportion_for_payer = {
		add = {
			desc = MYCENAEAN_LEAGUE_TRIBUTE_SHARE
			value = monthly_income_trade_and_tax
		}
		add = {
			desc = MYCENAEAN_LEAGUE_TRIBUTE_STORED_WEALTH
			value = gold
			multiply = 0.005
		}
		if = {
			limit = { mycenaean_league_has_palace_house = yes }
			add = {
				desc = MYCENAEAN_LEAGUE_PALACE_HOUSE_TRIBUTE
				value = 0.5
			}
		}
	}

	proportion_for_payee = {
		add = {
			desc = MYCENAEAN_LEAGUE_HIGH_WANAX_SHARE
			value = 1
		}
	}
}
