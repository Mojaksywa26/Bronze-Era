hussite_tribunal = {
	toggle = no
	army_only = yes
	
	map = yes
	
	hidden = {
		NOT = { is_situation_active = situation:hussite_wars }
		owner ?= { 
			OR = {
				NOT = { religion = religion:catholic }
				NOT = { can_see_situation = situation:hussite_wars }
				NOT = { is_at_war_with = c:BOH }
			}
		}
	}

	allow = {
		has_only_prisoners = no
		owner = { religious_influence > religious_influence_mild_bonus }
		in_combat = no
		is_exiled = no
		in_siege = no
		unit_location = { 
			custom_tooltip = {
				text = controller_is_not_owner_tt
				has_owner = yes
				controller ?= {
					this != prev.owner
					this = root.owner
				}
			}
			province = {
				NOT = { has_province_modifier = recent_tribunal_modifier }
			}
			is_province_capital = yes
			"religion_percentage(religion:hussite)" > 0.05
		}
	}
	
	start_effect = {
		owner = { add_religious_influence = religious_influence_mild_penalty }
		unit_location = {
			province = {

				add_province_modifier = {
					modifier = recent_tribunal_modifier
					years = modifier_duration_years_short
					mode = replace
				}
				every_location_in_province = {
					limit = {
						controller ?= {
							this != c:BOH
							OR = {
								this = root.owner
								AND = {
									is_at_war_with = c:BOH
									religion = religion:catholic
								}
							}
						}
					}
					ordered_pop = {
						limit = {
							owner = root.unit_location.owner
							religion = religion:hussite
						}
						max = 1
						order_by = pop_size
						check_range_bounds = no

						situation:hussite_wars = {
							change_variable = {
								name = heretics_eradicated_variable
								add = {
									value = prev.pop_size
									multiply = 50
								}
							}

							change_variable = {
								name = heretics_converted_variable
								add = {
									value = prev.pop_size
									multiply = 200
								}
							}
						}
		
						split_pop = {
							fraction = 0.2
							religion = religion:catholic
						}
		
						add_pop_size = {
							value = pop_size 
							multiply = -0.05
						}
					}
				}
			}
		}
	}

	ai_will_do = {
		add = {
			value = 100
			desc = BASE_VALUE
		}
	}
}
