mycenaean_league.1 = {
	type = country_event
	title = mycenaean_league.1.title
	desc = mycenaean_league.1.desc
	category = international_organization_event
	outcome = positive
	orphan = yes

	trigger = {
		is_leader_of_international_organization = international_organization:mycenaean_league
	}

	option = {
		name = mycenaean_league.1.a
		change_mycenaean_league_authority = { value = 5 }
		add_prestige = prestige_weak_bonus
	}
	option = {
		name = mycenaean_league.1.b
		change_mycenaean_league_authority = { value = 2 }
	}
}

	title = mycenaean_league.100.title

	major = yes
	orphan = yes

	trigger = {
		is_leader_of_international_organization = international_organization:mycenaean_league
